cmake_minimum_required(VERSION 3.16)

project(AudioCapturePlugin VERSION 1.0.0)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Enable Qt's MOC and automatic handling
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

# Find required Qt components
find_package(Qt6 REQUIRED COMPONENTS Core Qml Quick Multimedia)

# Find PulseAudio
find_package(PkgConfig REQUIRED)
pkg_check_modules(PULSEAUDIO REQUIRED libpulse libpulse-simple)

# Create QML plugin
qt6_add_qml_module(audiocaptureplugin
    URI "AudioCapture"
    VERSION 1.0
    SOURCES
        AudioCapture.cpp
        AudioCapture.h
    QML_FILES
        # No QML files needed
)

# Link libraries
target_link_libraries(audiocaptureplugin
    Qt6::Core
    Qt6::Qml
    Qt6::Quick
    Qt6::Multimedia
    ${PULSEAUDIO_LIBRARIES}
)

# Include directories
target_include_directories(audiocaptureplugin PRIVATE
    ${PULSEAUDIO_INCLUDE_DIRS}
)

# Compile definitions for PulseAudio
target_compile_definitions(audiocaptureplugin PRIVATE
    ${PULSEAUDIO_CFLAGS_OTHER}
)

# Install plugin to Qt's QML import path
install(TARGETS audiocaptureplugin
    DESTINATION ${CMAKE_INSTALL_PREFIX}/lib/qt6/qml/AudioCapture
)

# Install plugin metadata
qt6_qml_type_registration(audiocaptureplugin)
